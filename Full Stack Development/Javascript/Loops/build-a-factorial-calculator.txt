const num = 15;
function factorialCalculator(number){
  let result = 1;
  for(let index = 1; index<=number; index++){
    result = index*result;
  }
  return result;
}
const factorial = factorialCalculator(num);
const resultMsg = `Factorial of ${num} is ${factorial}`;
console.log(resultMsg);
